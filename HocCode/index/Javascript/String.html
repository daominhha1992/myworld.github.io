JavaScript String Methods

Method Công dụng Ví dụ ✓
lengthđộ dài chuỗi"abc".length // 3
toUpperCase()viết hoa"abc".toUpperCase() // "ABC"
toLowerCase()viết thường"ABC".toLowerCase() // "abc"
includes("x")kiểm tra chứa ký tự"hello".includes("he") // true
indexOf("x")vị trí đầu tiên"banana".indexOf("na") // 2
lastIndexOf("x")vị trí cuối cùng"banana".lastIndexOf("na") // 4
slice(a,b)cắt chuỗi từ a đến b-1"banana".slice(1,3) // "an"
substring(a,b)tương tự slice (ko nhận số âm)"banana".substring(1,3)
substr(a,n)cắt từ a, lấy n ký tự"banana".substr(1,2) // "an"
replace("x","y")thay thế"hello".replace("he","yo") // "yollo"
split(",")tách chuỗi thành mảng"a,b,c".split(",") // ["a","b","c"]
trim()xóa khoảng trắng đầu/đuôi" hi ".trim() // "hi"
charAt(i)ký tự tại vị trí i"abc".charAt(1) // "b"
charCodeAt(i)mã Unicode"A".charCodeAt(0) // 65
